// ============================= // 🔧 設定區 - 請修改以下變數 // ============================= const SPREADSHEET_ID = '1rCsP-0ijD0-aXmaax0sdlvgOSuN3zfkC5jwHklcY-Rs'; // Google 試算表 ID const SHEET_NAME = 'LSVTBIG'; // 工作表名稱 const FOLDER_ID = '1KyzJAniOeTXVAgekuQelA3l_5SHVqH-h'; // Google Drive 資料夾 ID（存放附件） // ============================= // 🌐 CORS Header 設定 // ============================= function setCORSHeaders(output) { return output .setHeader('Access-Control-Allow-Origin', '*') .setHeader('Access-Control-Allow-Methods', 'GET, POST, OPTIONS') .setHeader('Access-Control-Allow-Headers', 'Content-Type'); } // ============================= // 🌐 處理 GET 請求 // ── GET 同時處理：回傳頁面 & 資料查詢 ── // ============================= function doGet(e) { const action = e.parameter && e.parameter.action; // 若有 action 參數 → 當作 API 使用 if (action === 'getNotes') { const userId = e.parameter.userId || ''; const result = getNotes(userId); const output = ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); return setCORSHeaders(output); } // 否則回傳前端 HTML 頁面 return HtmlService.createHtmlOutputFromFile('index') .setTitle('📒 LIFF 筆記本') .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL); } // ============================= // 📮 處理 POST 請求（儲存筆記） // ============================= function doPost(e) { // ── 處理 OPTIONS preflight（CORS 預檢）── if (!e || !e.postData) { const output = ContentService .createTextOutput('') .setMimeType(ContentService.MimeType.TEXT); return setCORSHeaders(output); } try { const data = JSON.parse(e.postData.contents); const action = data.action || 'saveNote'; let result; switch (action) { case 'saveNote' : result = saveNote(data); break; case 'initSheet' : result = initSheet(); break; default : result = { success: false, message: '未知操作' }; } const output = ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); return setCORSHeaders(output); } catch (err) { const output = ContentService .createTextOutput(JSON.stringify({ success: false, message: err.message })) .setMimeType(ContentService.MimeType.JSON); return setCORSHeaders(output); } } // ============================= // 🗂️ 初始化工作表 // ============================= function initSheet() { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); let sheet = ss.getSheetByName(SHEET_NAME); if (sheet) { ss.deleteSheet(sheet); SpreadsheetApp.flush(); } sheet = ss.insertSheet(SHEET_NAME); const headers = [ '筆記ID', '建立時間', '使用者ID', '使用者名稱', '標題', '內容', '附件名稱', '附件連結' ]; const headerRange = sheet.getRange(1, 1, 1, headers.length); headerRange.setValues([headers]); headerRange .setFontWeight('bold') .setFontColor('#FFFFFF') .setBackground('#06C755') .setHorizontalAlignment('center') .setVerticalAlignment('middle') .setFontSize(11); sheet.setFrozenRows(1); const colWidths = [160, 160, 200, 130, 200, 300, 160, 260]; colWidths.forEach((width, i) => sheet.setColumnWidth(i + 1, width)); sheet.setRowHeight(1, 36); sheet.getRange('A:A').setFontFamily('Courier New'); sheet.getRange('B:B').setNumberFormat('@'); sheet.getRange('H:H').setWrap(true); const protection = sheet.getRange(1, 1, 1, headers.length).protect(); protection.setDescription('標題列保護'); protection.setWarningOnly(true); SpreadsheetApp.flush(); Logger.log('✅ 工作表初始化完成：' + SHEET_NAME); return { success: true, message: '工作表初始化完成！', sheetName: SHEET_NAME }; } // ============================= // 💾 儲存筆記 // ============================= function saveNote(data) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { initSheet(); sheet = ss.getSheetByName(SHEET_NAME); } let fileUrl = ''; let fileName = ''; if (data.fileData && data.fileName) { const fileResult = uploadFileToDrive(data.fileData, data.fileName, data.fileType); fileUrl = fileResult.url; fileName = fileResult.name; } const noteId = 'NOTE_' + new Date().getTime(); const timestamp = Utilities.formatDate(new Date(), 'Asia/Taipei', 'yyyy-MM-dd HH:mm:ss'); sheet.appendRow([ noteId, timestamp, data.userId || 'unknown', data.userName || '匿名使用者', data.title || '(無標題)', data.content || '', fileName, fileUrl ]); const lastRow = sheet.getLastRow(); const rowBg = lastRow % 2 === 0 ? '#F0FFF4' : '#FFFFFF'; sheet.getRange(lastRow, 1, 1, 8).setBackground(rowBg); sheet.autoResizeColumns(1, 8); return { success: true, noteId: noteId, message: '筆記儲存成功！', fileUrl: fileUrl }; } // ============================= // 📂 上傳檔案到 Google Drive // ============================= function uploadFileToDrive(base64Data, fileName, mimeType) { const folder = DriveApp.getFolderById(FOLDER_ID); const decoded = Utilities.base64Decode(base64Data.split(',')[1] || base64Data); const blob = Utilities.newBlob(decoded, mimeType || 'application/octet-stream', fileName); const file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); return { url: file.getDownloadUrl(), name: file.getName(), id: file.getId() }; } // ============================= // 📋 取得筆記列表 // ============================= function getNotes(userId) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) return { success: true, notes: [] }; const rows = sheet.getDataRange().getValues(); const notes = []; for (let i = 1; i < rows.length; i++) { const row = rows[i]; if (!userId || row[2] === userId) { notes.push({ noteId : row[0], time : row[1], userId : row[2], userName : row[3], title : row[4], content : row[5], fileName : row[6], fileUrl : row[7] }); } } notes.reverse(); return { success: true, notes: notes }; }
